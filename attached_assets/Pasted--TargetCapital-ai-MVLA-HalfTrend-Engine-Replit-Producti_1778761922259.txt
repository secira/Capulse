# 13. REPLIT-READY HALF TREND CODE

```python
import numpy as np
import pandas as pd


# ======================================
# Pine-style RMA
# ======================================
def pine_rma(series: pd.Series, length: int):

    alpha = 1 / length

    rma = np.zeros(len(series))

    rma[0] = series.iloc[0]

    for i in range(1, len(series)):
        rma[i] = alpha * series.iloc[i] + (1 - alpha) * rma[i - 1]

    return pd.Series(rma, index=series.index)


# ======================================
# ATR
# ======================================
def calculate_atr(df, length=14):

    high = df["high"]
    low = df["low"]
    close = df["close"]

    prev_close = close.shift(1)

    tr = pd.concat([
        high - low,
        (high - prev_close).abs(),
        (low - prev_close).abs()
    ], axis=1).max(axis=1)

    return pine_rma(tr, length)


# ======================================
# HalfTrend Engine
# ======================================
def halftrend_engine(
    df,
    amplitude=4,
    atr_length=14,
    channel_deviation=1.5
):

    data = df.copy().reset_index(drop=True)

    high = data["high"]
    low = data["low"]
    close = data["close"]

    atr = calculate_atr(data, atr_length)
    atr_half = atr / 2

    n = len(data)

    trend = np.zeros(n, dtype=int)

    halftrend = np.full(n, np.nan)

    buy_exit = np.full(n, False)
    sell_exit = np.full(n, False)

    stability_score = np.full(n, 0.0)

    rolling_high = high.rolling(amplitude).max()
    rolling_low = low.rolling(amplitude).min()

    avg_range_5 = (high - low).rolling(5).mean()
    current_range = high - low

    volatility_expansion = current_range > (avg_range_5 * 1.2)

    current_trend = 0

    for i in range(amplitude, n):

        bullish_structure = close.iloc[i] > rolling_high.iloc[i - 1]
        bearish_structure = close.iloc[i] < rolling_low.iloc[i - 1]

        # -----------------------------
        # Bullish Trend
        # -----------------------------
        if bullish_structure:

            current_trend = 0

            halftrend[i] = rolling_low.iloc[i] - atr_half.iloc[i]

            if trend[i - 1] == 1:
                buy_exit[i] = False

        # -----------------------------
        # Bearish Trend
        # -----------------------------
        elif bearish_structure:

            current_trend = 1

            halftrend[i] = rolling_high.iloc[i] + atr_half.iloc[i]

            if trend[i - 1] == 0:
                sell_exit[i] = False

        else:

            current_trend = trend[i - 1]
            halftrend[i] = halftrend[i - 1]

        trend[i] = current_trend

        # -----------------------------
        # Exit Conditions
        # -----------------------------

        # Exit CALL
        if trend[i] == 0:
            if close.iloc[i] < halftrend[i]:
                buy_exit[i] = True

        # Exit PUT
        if trend[i] == 1:
            if close.iloc[i] > halftrend[i]:
                sell_exit[i] = True

        # -----------------------------
        # Stability Score
        # -----------------------------
        score = 0

        # Trend persistence
        if i >= 3:
            if trend[i] == trend[i - 1] == trend[i - 2]:
                score += 10

        # Volatility expansion
        if volatility_expansion.iloc[i]:
            score += 15

        # Price distance from HalfTrend
        distance = abs(close.iloc[i] - halftrend[i])

        if distance > atr.iloc[i] * 0.5:
            score += 10

        # Strong candle
        if current_range.iloc[i] > avg_range_5.iloc[i]:
            score += 10

        stability_score[i] = min(score, 100)

    result = data.copy()

    result["trend"] = trend
    result["halftrend"] = halftrend

    result["buy_exit"] = buy_exit
    result["sell_exit"] = sell_exit

    result["stability_score"] = stability_score

    result["volatility_expansion"] = volatility_expansion

    return result
```
